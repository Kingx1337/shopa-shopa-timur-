        // API ключ Gemini - оставляем пустым, как того требуют правила окружения

        const tradingPairs = [

            { symbol: "ETHUSDT", displayName: "ETH/USDT", baseAsset: "ETH" },
            { symbol: "SOLUSDT", displayName: "SOL/USDT", baseAsset: "SOL" },

            { symbol: "XRPUSDT", displayName: "XRP/USDT", baseAsset: "XRP" },

        ];

        let activePair = tradingPairs[0];
        let marketData = {};

        window.addEventListener('DOMContentLoaded', () => {

                const query = e.target.value.toLowerCase();

                cards.forEach(card => {
                    const text = card.textContent.toLowerCase();
                    if (text.includes(query)) {
                        card.classList.remove('hidden');
                    } else {
                        card.classList.add('hidden');
                    }
                });
            });

        async function initApp() {
            renderPairsList();
            await refreshMarketPrices();
            await selectPair(activePair.symbol);
            
            // Регулярное обновление котировок в фоне
            setInterval(async () => {
                await refreshMarketPrices();
                if (marketData[activePair.symbol]) {
                    const freshData = marketData[activePair.symbol];
                    document.getElementById('active-pair-price').innerText = `$${parseFloat(freshData.price).toLocaleString(undefined, { minimumFractionDigits: 2, maximumFractionDigits: 4 })}`;
                }
            }, 8000);
        }

        // Вывод списка торговых пар
        function renderPairsList() {
            const listEl = document.getElementById('trading-pairs-list');
            listEl.innerHTML = '';
            
            tradingPairs.forEach(pair => {
                const isSelected = pair.symbol === activePair.symbol;
                const activeClass = isSelected ? 'bg-blue-600/10 border-blue-500/50' : 'bg-[#121620]/40 border-gray-800/40 hover:bg-[#121620]/80 hover:border-gray-800';
                const pairData = marketData[pair.symbol] || { price: "0.00", change: "0.00", volume: "0" };
                const isPositive = parseFloat(pairData.change) >= 0;
                const changeClass = isPositive ? 'text-emerald-400' : 'text-rose-500';
                
                const card = document.createElement('div');
                card.className = `p-3 rounded-xl border ${activeClass} cursor-pointer transition flex items-center justify-between group`;
                card.id = `pair-card-${pair.symbol}`;
                card.onclick = () => selectPair(pair.symbol);
                
                card.innerHTML = `
                    <div class="flex items-center gap-2.5">
                        <div class="w-8 h-8 rounded-lg bg-gray-800 flex items-center justify-center font-bold text-xs text-gray-300 group-hover:bg-blue-600/20 group-hover:text-blue-400 transition">
                            ${pair.baseAsset}
                        </div>
                        <div>
                            <span class="font-bold text-sm block text-gray-200">${pair.displayName}</span>
                            <span class="text-[10px] text-gray-500 font-mono">Объем: ${parseFloat(pairData.volume || 0).toLocaleString(undefined, { maximumFractionDigits: 0 })}</span>
                        </div>
                    </div>
                    <div class="text-right">
                        <span class="font-mono font-bold text-sm block text-gray-200">$${parseFloat(pairData.price || 0).toLocaleString(undefined, { minimumFractionDigits: 2, maximumFractionDigits: 2 })}</span>
                        <span class="text-[11px] font-semibold font-mono flex items-center justify-end gap-0.5 ${changeClass}">
                            ${isPositive ? '+' : ''}${pairData.change}%
                        </span>
                    </div>
                `;
                listEl.appendChild(card);
            });
        }

        // Обновление цен через API Binance
        async function refreshMarketPrices() {
            try {
                const response = await fetch('https://api.binance.com/api/v3/ticker/24hr');
                if (!response.ok) throw new Error("API Limit");
                const data = await response.json();
                
                const targetSymbols = tradingPairs.map(p => p.symbol);
                const filtered = data.filter(item => targetSymbols.includes(item.symbol));
                
                filtered.forEach(item => {
                    marketData[item.symbol] = {
                        price: item.lastPrice,
                        change: parseFloat(item.priceChangePercent).toFixed(2),
                        volume: item.quoteVolume,
                        high: item.highPrice,
                        low: item.lowPrice
                    };
                });
                
                document.getElementById('last-update-time').innerText = new Date().toLocaleTimeString();
                renderPairsList();
            } catch (err) {
                console.warn("API Binance недоступно, симулируем котировки:", err);
                simulateMarketDataFallback();
            }
        }

        // Симуляция на случай блокировок или оффлайн-режима
        function simulateMarketDataFallback() {
            tradingPairs.forEach(pair => {
                if (!marketData[pair.symbol]) {
                    let basePrice = 55000;
                    if (pair.symbol.includes("ETH")) basePrice = 3100;
                    if (pair.symbol.includes("SOL")) basePrice = 145;
                    if (pair.symbol.includes("BNB")) basePrice = 580;
                    if (pair.symbol.includes("ADA")) basePrice = 0.50;
                    
                    marketData[pair.symbol] = {
                        price: basePrice * (1 + (Math.random() - 0.5) * 0.01),
                        change: ((Math.random() - 0.4) * 3).toFixed(2),
                        volume: Math.random() * 45000000,
                        high: basePrice * 1.02,
                        low: basePrice * 0.98
                    };
                } else {
                    const current = parseFloat(marketData[pair.symbol].price);
                    marketData[pair.symbol].price = (current * (1 + (Math.random() - 0.5) * 0.0015)).toFixed(4);
                }
            });
            document.getElementById('last-update-time').innerText = new Date().toLocaleTimeString() + " (Демо)";
            renderPairsList();
        }

        // Переключение торгового актива
        async function selectPair(symbol) {
            const pair = tradingPairs.find(p => p.symbol === symbol);
            if (!pair) return;
            activePair = pair;

        function calcRSI(closes, period = 14) {
            let gains = 0;
            let losses = 0;
            for (let i = closes.length - period; i < closes.length; i++) {
                let diff = closes[i] - closes[i - 1];
                if (diff > 0) gains += diff;
                else losses -= diff;
            }
            if (losses === 0) return 100;
            let rs = (gains / period) / (losses / period);
            return 100 - (100 / (1 + rs));
        }

        function calcEMA(closes, period) {
            let k = 2 / (period + 1);
            let ema = closes[closes.length - period];
            for (let i = closes.length - period + 1; i < closes.length; i++) {
                ema = (closes[i] * k) + (ema * (1 - k));
            }
            return ema;
        }

        function calcMACD(closes) {
            const ema12 = calcEMA(closes, 12);
            const ema26 = calcEMA(closes, 26);
            const macdLine = ema12 - ema26;
            const signalLine = macdLine * 0.75; 
            return { value: macdLine, signal: signalLine };
        }

        // Отрисовка графиков свечей на холсте
        function drawChart() {
            const canvas = document.getElementById('candlestick-chart');
            if (!canvas) return;
            const ctx = canvas.getContext('2d');
            
            const rect = canvas.parentNode.getBoundingClientRect();
            canvas.width = rect.width * window.devicePixelRatio;
            canvas.height = rect.height * window.devicePixelRatio;
            ctx.scale(window.devicePixelRatio, window.devicePixelRatio);
            
            const width = rect.width;
            const height = rect.height;
            
            ctx.fillStyle = '#0F131C';
            ctx.fillRect(0, 0, width, height);
            
            if (activeKlines.length === 0) return;
            
            const candlesCount = Math.min(activeKlines.length, 50);
            const visibleCandles = activeKlines.slice(-candlesCount);
            
            let maxPrice = Math.max(...visibleCandles.map(k => k.high));
            let minPrice = Math.min(...visibleCandles.map(k => k.low));
            
            const pricePadding = (maxPrice - minPrice) * 0.12;
            maxPrice += pricePadding;
            minPrice -= pricePadding;
            
            const maxVolume = Math.max(...visibleCandles.map(k => k.volume));
            
            const paddingRight = 65;
            const paddingBottom = 20;
            const chartWidth = width - paddingRight;
            const chartHeight = height - paddingBottom;
            const candleWidth = chartWidth / candlesCount;
            
            ctx.strokeStyle = '#1D2533';
            ctx.lineWidth = 1;
            ctx.font = '10px "JetBrains Mono", monospace';
            ctx.fillStyle = '#64748B';
            
            // Отрисовка сетки цены
            const gridLinesCount = 5;
            for (let i = 0; i < gridLinesCount; i++) {
                const ratio = i / (gridLinesCount - 1);
                const price = maxPrice - (ratio * (maxPrice - minPrice));
                const y = ratio * (chartHeight - 40) + 20;
                
                ctx.beginPath();
                ctx.moveTo(0, y);
                ctx.lineTo(chartWidth, y);
                ctx.stroke();
                
                ctx.fillText(`$${price.toLocaleString(undefined, { maximumFractionDigits: 2 })}`, chartWidth + 5, y + 4);
            }
            
            // Отрисовка японских свечей
            visibleCandles.forEach((candle, idx) => {
                const x = idx * candleWidth + (candleWidth * 0.15);
                const cWidth = candleWidth * 0.7;
                
                const yOpen = ((maxPrice - candle.open) / (maxPrice - minPrice)) * (chartHeight - 40) + 20;
                const yClose = ((maxPrice - candle.close) / (maxPrice - minPrice)) * (chartHeight - 40) + 20;
                const yHigh = ((maxPrice - candle.high) / (maxPrice - minPrice)) * (chartHeight - 40) + 20;
                const yLow = ((maxPrice - candle.low) / (maxPrice - minPrice)) * (chartHeight - 40) + 20;
                
                const isGreen = candle.close >= candle.open;
                const color = isGreen ? '#10B981' : '#EF4444';
                
                ctx.strokeStyle = color;
                ctx.lineWidth = 1.5;
                ctx.beginPath();
                ctx.moveTo(x + cWidth/2, yHigh);
                ctx.lineTo(x + cWidth/2, yLow);
                ctx.stroke();
                
                ctx.fillStyle = color;
                ctx.fillRect(x, Math.min(yOpen, yClose), cWidth, Math.max(Math.abs(yOpen - yClose), 1.5));
                
                // Столбцы объемов внизу графика
                const volHeight = (candle.volume / maxVolume) * 35;
                ctx.fillStyle = isGreen ? 'rgba(16, 185, 129, 0.15)' : 'rgba(239, 68, 68, 0.15)';
                ctx.fillRect(x, chartHeight - volHeight, cWidth, volHeight);
            });
            
            // Текущая цена (пунктирная линия)
            const lastCandle = visibleCandles[visibleCandles.length - 1];
            const lastPriceY = ((maxPrice - lastCandle.close) / (maxPrice - minPrice)) * (chartHeight - 40) + 20;
            
            ctx.strokeStyle = '#3B82F6';
            ctx.setLineDash([4, 4]);
            ctx.beginPath();
            ctx.moveTo(0, lastPriceY);
            ctx.lineTo(chartWidth, lastPriceY);
            ctx.stroke();
            ctx.setLineDash([]);
            
            ctx.fillStyle = '#3B82F6';
            ctx.fillRect(chartWidth, lastPriceY - 8, paddingRight, 16);
            ctx.fillStyle = '#FFFFFF';
            ctx.font = 'bold 9px "JetBrains Mono", monospace';
            ctx.fillText(lastCandle.close.toFixed(2), chartWidth + 4, lastPriceY + 4);
        }

        function cleanJsonString(rawText) {
            let clean = rawText.trim();
            if (clean.startsWith("```json")) {
                clean = clean.substring(7);
            } else if (clean.startsWith("```")) {
                clean = clean.substring(3);
            }
            if (clean.endsWith("```")) {
                clean = clean.substring(0, clean.length - 3);
            }
            return clean.trim();
        }

        async function callGeminiWithRetry(query, systemInstruction, retries = 5, delay = 1000) {
            const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
            const payload = {
                contents: [{ parts: [{ text: query }] }],
                systemInstruction: { parts: [{ text: systemInstruction }] },
                generationConfig: { responseMimeType: "application/json" }
            };

            for (let i = 0; i < retries; i++) {
                try {
                    const response = await fetch(url, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(payload)
                    });

                    if (response.ok) {
                        const result = await response.json();
                        return result.candidates?.[0]?.content?.parts?.[0]?.text;
                    }
                    
                    if (response.status === 429 || response.status >= 500) {
                        await new Promise(resolve => setTimeout(resolve, delay));
                        delay *= 2;
                        continue;
                    }
                    throw new Error(`Gemini API HTTP Error status ${response.status}`);
                } catch (err) {
                    if (i === retries - 1) throw err;
                    await new Promise(resolve => setTimeout(resolve, delay));
                    delay *= 2;
                }
            }
            throw new Error("Failed after all retries");
        }
